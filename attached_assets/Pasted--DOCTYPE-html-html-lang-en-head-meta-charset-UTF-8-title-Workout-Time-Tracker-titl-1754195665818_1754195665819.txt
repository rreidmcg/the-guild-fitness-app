  <div class="modal-content">
    <p id="modalText">Your workout time seems off. Adjust it if needed:</p>
    <input type="number" id="manualTime" min="1" />
    <button onclick="confirmTime()">Confirm</button>

<script>
  const estimatedTime = 15;
  let startTime = null;
  let endTime = null;

  function startWorkout() {
    startTime = new Date();
    alert('Workout started!');
  }

  function endWorkout() {
    if (!startTime) {
      alert('Start the workout first!');
      return;
    }
    endTime = new Date();
    const elapsedMs = endTime - startTime;
    const actualMin = elapsedMs / 60000;
    const low = estimatedTime * 0.5;
    const high = estimatedTime * 1.5;

    if (actualMin < low || actualMin > high) {
      showModal(Math.round(actualMin));
    } else {
      saveWorkout(Math.round(actualMin));
    }
  }

  function showModal(actual) {
    document.getElementById('timeModal').style.display = 'flex';
    document.getElementById('manualTime').value = actual;
  }

  function confirmTime() {
    const time = parseInt(document.getElementById('manualTime').value);
    if (!isNaN(time)) {
      saveWorkout(time);
      document.getElementById('timeModal').style.display = 'none';
    }
  }

  function saveWorkout(time) {
    console.log(`Workout saved with time: ${time} mins`);
    alert(`Workout saved with time: ${time} mins`);
    // You can now push this to localStorage, DB, or analytics system
  }
</script>
